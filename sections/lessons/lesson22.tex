\subsection{Занятие 22. 2024.11.28}
\begin{newwords}\hfill
	\begin{enumerate}
	\item 号 - hào - 1) число, номер\\
	2) дата
	\begin{example}\hfill\break
		电话号(码) - номер телефона.
	\end{example}
	\item 纸 - zhǐ - бумага
	\item 报 - bào - морфема "информировать"
	\item 报纸 - bàozhǐ - газета
	\item 好用 - hǎoyòng - удобен в использовании
	\item 喜欢 - xǐhuan - нравиться
	\begin{example}\hfill\break
		我喜欢学习汉语。 - Мне нравится изучать китайский язык.
	\end{example}
	\item 玩 - wán - чаще всего используется с эризацией:\\
	玩儿 - wánr\\
	1) Играть\\
	2) Развлекаться, проводить время
	\begin{example}\hfill\break
		玩(儿)电脑 - играть на компьютере.\\
		玩(儿)手机 - залипать в телефон.\\
		他去北京玩儿。 - Он едет в Пекин не по делу.
	\end{example}
	\item 游戏 - 游戏 - игра
	\item 第 - dì - ставится перед количественным числительным и образует порядковое числительное (какой?)
	\begin{example}\hfill\break
		第六课 - шестой урок.\\
		第二个 - второй. (Не 两!)
	\end{example}
	\item 认识 - rènshi - быть знакомым (знать кого-то)
	\begin{example}\hfill\break
		我认识他。 - Я его(о нем) знаю.
	\end{example}
	\item 高兴 - gāoxìng - рад
	\begin{example}\hfill\break
		我很高兴认识你。 - Рад познакомиться
	\end{example}
	\begin{newgraphem}
		虫 - chóng - насекомое
	\end{newgraphem}
	\item 触屏 - chùpíng - сенсорный экран
	\begin{example}\hfill\break
		触屏手机 - Сотовый телефон с сенсорным экраном.\\
		按键手机 - Кнопочный телефон.
	\end{example}
	\item 屏幕 - píngmù - экран, дисплей
	\begin{example}\hfill\break
		屏幕大的手机 - телефон с большим экраном
	\end{example}
	\item 种 - zhǒng - вид, разновидность
	\begin{example}\hfill\break
		这种水果 - этот вид фруктов
	\end{example}
	\end{enumerate}
\end{newwords}
