\begin{newgraphems}
	\item 果 - guǒ - плод
	\begin{example}
		水果 - фрукт
	\end{example}
\end{newgraphems}
\textbf{Слова, обозначающие степень признака}
\begin{itemize}
	\item 特别 - tèbié - особенно (очень)
	\begin{example}\hfill\break
		汉语特别难。 - Китайский язык особенно трудный.
	\end{example}
	\item 好 + verb - положительный с точки зрения ...\\ (трансформация глагола в прилагательное)
	\begin{example}\hfill\break
		好喝 - вкусный по питьевым ощущениям\\
		中国茶很特别好喝 - Китайский чай очень вкусный.\\
		好吃 - вкусный (о еде, 吃 - chī - кушать)\\
		好学 - легко дается в плане учебы\\
		好看 - визуально привлекательный
	\end{example}
	\item 听 - tīng - слушать
	\begin{example}\hfill\break
		好听 - приятный на слух, благозвучный
	\end{example}
	\item 难 (nán) + verb - трудный с точки зрения ... (трансформация глагола в прилагательное)
	\begin{example}\hfill\break
		难学 - труден в изучении\\
		酒 - jiǔ - морфема "алкоголь"\\
		白 - bái - белый\\
		中国白酒难喝。 - Китайскую водку пить невозможно.
	\end{example}
\end{itemize}
\begin{newwords}\hfill
	\begin{enumerate}
		\item 错 - cuò - ошибаться
		\item 错误 - cuòwù - ошибка
		\begin{example}
			语法错误 - грамматическая ошибка
		\end{example}
		\item 课 - kè - урок/занятие
		\item 上课 - shàngkè - быть на уроке
		\item 得 - děi - надо, нужно (нет других вариантов)
		\begin{example}
			我得上课。 - Мне нужно быть на уроке
		\end{example}
		\item 走 - zǒu - идти(пешком)/уходить
		\item 先 - xiān - морфема "сначала"
		\begin{example}
			先走 - сейчас пошел (я иду?/пойдем?)
		\end{example}
		\item 啦 - la - конечная частица\\
		(удовлетворенность, становление действия,\\ завершенность действия)
		\begin{example}
			先走啦。 - Ну я пошел?
		\end{example}
		\item 再 - zài - еще (говоря про то же самое действие)
		\item 再见
	\end{enumerate}
\end{newwords}
