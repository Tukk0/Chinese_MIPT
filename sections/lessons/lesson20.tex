\textbf{Слова, обозначающие степень признака}
\begin{itemize}
	\item 特别 - tèbié - особенно (очень)
	\begin{example}\hfill\break
		汉语特别难。 - Китайский язык особенно трудный.
	\end{example}
	\item 好 + verb - положительный с точки зрения ...\\ (трансформация глагола в прилагательное)
	\begin{example}\hfill\break
		好喝 - вкусный по питьевым ощущениям\\
		中国茶很特别好喝 - Китайский чай очень вкусный.\\
		好吃 - вкусный (о еде, 吃 - chī - кушать)\\
		好学 - легко дается в плане учебы\\
		好看 - визуально привлекательный
	\end{example}
	\item 听 - tīng - слушать
	\begin{example}\hfill\break
		好听 - приятный на слух, благозвучный
	\end{example}
	\item 难 (nán) + verb - трудный с точки зрения ... (трансформация глагола в прилагательное)
	\begin{example}\hfill\break
		难学 - труден в изучении\\
		酒 - jiǔ - морфема "алкоголь"\\
		白 - bái - белый\\
		中国白酒难喝。 - Китайскую водку пить невозможно.
	\end{example}
\end{itemize}
