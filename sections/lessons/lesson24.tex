\subsection{Занятие 24. 2024.12.03}
\begin{newwords}
	\begin{enumerate}\hfill
		\item 年 - nián - год
		\begin{example}
			新年 - Новый Год
		\end{example}
		\item 月 - yuè - месяц
		\item 号 - hào - день
		\item 几 - jǐ - вопросительное слово для дат (сколько)
		\item 家 - jiā - 1) Семья (дом, семейный очаг)\\
		2) Суффикс существительных со значением\\
		"специалист в области..."
		\begin{example}\hfill\break
			妈妈在家看书。 - Мама читает дома книгу.\\
			画家 - художник\\
			科学家 - ученый
		\end{example}
		\item 鼠 - shǔ - мышь
		\item 鼠标 - shǔbiāo - компьютерная мышь
		\begin{newgraphem}
			广 - yǎn - навес
		\end{newgraphem}
		\item 店 - diàn - конечная морфема "точка реализации"
		\begin{example}
			书店 - книжный магазин
		\end{example}
		\item 商 - shāng - морфема "коммерция"
		\item 商店 - shāngdiàn - магазин
		\item 开 - kāi - морфема "открывать"
		\item 开心 - kāixīn - довольный
		\item 天 - tiān - день (время суток)
		\item 今天 - jīntiān - сегодня
		\begin{example}
			我今天很开心。 - Я сегодня доволен.
		\end{example}
		\item 华为 - huáwéi - Huawei
		\item 公司 - gōngsī - компания
		\begin{example}\hfill\break
			华为公司 - Компания Huawei\\
			他在华为公司工作。 - Он работает в компании Huawei.
		\end{example}
	\end{enumerate}
\end{newwords}
