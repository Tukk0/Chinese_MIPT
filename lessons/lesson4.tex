\begin{newwords}\hfill
	\begin{enumerate}
		\item 今天 - jīntiān - сегодня
		\item 明天 - míngtiān - завтра
		\item 昨天 - zuótiān - вчера\\
		天 - tiān - небо, день
		\begin{example}
			明天晚上 - завтра вечером.
		\end{example}
		\item 测 - cè - морфема "измерять"
		\item 试 - shì - морфема "пробовать"
		\item 测试 - cèshì - тест, экзамен
		\item 担心 - dānxīn - беспокоиться
		\item 别 - bié - повелительное "не ..." к кому-либо
		\begin{example}\hfill\break
			别担心 - не беспокойтесь\\
			别说话 - не разговаривайте
		\end{example}
		\item 聪明 - cōngmíng - умный
		\item 通过 - tōngguò - пройти, сдать (тест/экзамен)
		\item 会 - huì - выражает возможность совершения действия в будущем
		\begin{example}\hfill\break
			我明天右侧是。 我不会通过。 У меня завтра тест. Я не смогу его сдать.
		\end{example}
		\item 解答 - jiědá - отвечать на вопросы теста
		\item 怎么 - zěnme - как, каким образом?
		\begin{example}\hfill\break
			汉字怎么写 - Как пишется иероглиф?
		\end{example}
	\end{enumerate}
\end{newwords}
