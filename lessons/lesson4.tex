\subsection{Занятие 4. 2025 年 02 月 13 号}
\begin{newgraphems} + повторение\hfill
	\begin{itemize}
		\item 手 (扌) - shǒu - рука
		\item 角 - jiǎo - рог
	\end{itemize}
\end{newgraphems}
\begin{newwords}\hfill
	\begin{enumerate}
		\item 今天 - jīntiān - сегодня
		\item 明天 - míngtiān - завтра
		\item 昨天 - zuótiān - вчера\\
		天 - tiān - небо, день
		\begin{example}
			明天晚上 - завтра вечером.
		\end{example}
		\item 测 - cè - морфема "измерять"
		\item 试 - shì - морфема "пробовать"
		\item 测试 - cèshì - тест, экзамен
		\item 担心 - dānxīn - беспокоиться
		\item 别 - bié - повелительное "не ..." к кому-либо
		\begin{example}\hfill\break
			别担心 - не беспокойтесь\\
			别说话 - не разговаривайте
		\end{example}
		\item 聪明 - cōngmíng - умный
		\item 通过 - tōngguò - пройти, сдать (тест/экзамен)
		\item 会 - huì - выражает возможность совершения действия в будущем
		\begin{example}\hfill\break
			我明天右侧是。 我不会通过。 У меня завтра тест. Я не смогу его сдать.
		\end{example}
		\item 解答 - jiědá - отвечать на вопросы теста
		\item 怎么 - zěnme - как, каким образом?
		\begin{example}\hfill\break
			汉字怎么写 - Как пишется иероглиф?\\
			怎么做实验 - Как проводить эксперимент?
		\end{example}
		\item 准备 - zhǔnbèi - готовиться (к тестам, экзаменам)
		\begin{example}\hfill\break
			我要准备汉语测试。 - Мне нужно готовиться к тесту по китайскому
		\end{example}
		\item Глагол + 一下 (yīxià) - непродолжительность действия
		\begin{example}
			我可以不可以看一下？ - Можно мне посмотреть?\\
			可以问一下？ - Можно спросить?\\
			认识一下 - познакомиться\\
			准备一下 - подготовиться
		\end{example}
		\item 一定 - yídìng - 1) определенно, наверняка    2) обязательно
		\begin{example}\hfill\break
			你一定通过 - ты несомненно сдашь тест\\
			你一定要来 - тебе обязательно нужно приехать
		\end{example}
		\item 所以 - suǒyǐ - поэтому
		\item 加 - jiā - добавить, прибавить
		\begin{example}
			二加二 - 2 + 2
		\end{example}
		\item 油 - yóu - масло
		\item 努力 - nǔlì - старательный, усердный
		\item 今年 - jīnnián - этот год
		\item 明天 - míngtiān - следующий год
		\item 去年 - qùnián - прошлый год
	\end{enumerate}
\end{newwords}
